### 2.2 字段类型说明

| type | 说明 | 特有属性 |
|------|------|---------|
| **input** | 输入框 | placeholder, maxLength |
| **textarea** | 文本域 | placeholder, maxLength, rows |
| **select** | 下拉选择 | options |
| **radio** | 单选框 | options |
| **checkbox** | 复选框 | options |
| **date** | 日期选择 | format, minDate, maxDate |
| **number** | 数字输入 | min, max, step |
| **switch** | 开关 | checkedValue, uncheckedValue |

## 三、前端渲染（Vue 示例）

### 3.1 基础渲染组件

```vue
<template>
  <el-form :model="formData" :rules="rules">
    <div v-for="field in schema.fields" :key="field.name">
      
      <!-- 输入框 -->
      <el-form-item 
        v-if="field.type === 'input'"
        :label="field.label"
        :prop="field.name">
        <el-input 
          v-model="formData[field.name]"
          :placeholder="field.placeholder"
          :maxlength="field.maxLength"/>
      </el-form-item>
      
      <!-- 文本域 -->
      <el-form-item 
        v-else-if="field.type === 'textarea'"
        :label="field.label"
        :prop="field.name">
        <el-input 
          type="textarea"
          v-model="formData[field.name]"
          :placeholder="field.placeholder"
          :rows="field.rows || 4"/>
      </el-form-item>
      
      <!-- 下拉选择 -->
      <el-form-item 
        v-else-if="field.type === 'select'"
        :label="field.label"
        :prop="field.name">
        <el-select v-model="formData[field.name]">
          <el-option
            v-for="(label, value) in field.options"
            :key="value"
            :label="label"
            :value="value"/>
        </el-select>
      </el-form-item>
      
      <!-- 单选框 -->
      <el-form-item 
        v-else-if="field.type === 'radio'"
        :label="field.label"
        :prop="field.name">
        <el-radio-group v-model="formData[field.name]">
          <el-radio
            v-for="(label, value) in field.options"
            :key="value"
            :label="value">{{ label }}</el-radio>
        </el-radio-group>
      </el-form-item>
      
      <!-- 复选框 -->
      <el-form-item 
        v-else-if="field.type === 'checkbox'"
        :label="field.label"
        :prop="field.name">
        <el-checkbox-group v-model="formData[field.name]">
          <el-checkbox
            v-for="(label, value) in field.options"
            :key="value"
            :label="value">{{ label }}</el-checkbox>
        </el-checkbox-group>
      </el-form-item>
      
      <!-- 日期选择 -->
      <el-form-item 
        v-else-if="field.type === 'date'"
        :label="field.label"
        :prop="field.name">
        <el-date-picker
          v-model="formData[field.name]"
          type="date"
          :format="field.format || 'yyyy-MM-dd'"
          :disabled-date="disabledDate"/>
      </el-form-item>
      
      <!-- 数字输入 -->
      <el-form-item 
        v-else-if="field.type === 'number'"
        :label="field.label"
        :prop="field.name">
        <el-input-number 
          v-model="formData[field.name]"
          :min="field.min"
          :max="field.max"
          :step="field.step || 1"/>
      </el-form-item>
      
    </div>
    
    <el-form-item>
      <el-button type="primary" @click="submit">提交</el-button>
    </el-form-item>
  </el-form>
</template>

<script>
export default {
  props: {
    schema: Object // JSON Schema
  },
  data() {
    return {
      formData: {},
      rules: {}
    };
  },
  created() {
    this.initFormData();
    this.initRules();
  },
  methods: {
    initFormData() {
      this.schema.fields.forEach(field => {
        this.formData[field.name] = field.defaultValue || null;
      });
    },
    initRules() {
      const rules = {};
      this.schema.fields.forEach(field => {
        const fieldRules = [];
        
        // 必填验证
        if (field.required) {
          fieldRules.push({ required: true, message: `${field.label}不能为空` });
        }
        
        // 长度验证
        if (field.maxLength) {
          fieldRules.push({ 
            max: field.maxLength, 
            message: `${field.label}长度不能超过${field.maxLength}` 
          });
        }
        
        // 范围验证
        if (field.min && field.max) {
          fieldRules.push({ 
            type: 'number', 
            min: field.min, 
            max: field.max, 
            message: `${field.label}必须在${field.min}到${field.max}之间` 
          });
        }
        
        // 正则验证
        if (field.validate?.pattern) {
          fieldRules.push({ 
            pattern: new RegExp(field.validate.pattern), 
            message: `${field.label}格式不正确` 
          });
        }
        
        if (fieldRules.length > 0) {
          rules[field.name] = fieldRules;
        }
      });
      this.rules = rules;
    },
    disabledDate(date) {
      // 日期禁用逻辑
      return false;
    },
    submit() {
      this.$refs.form.validate(valid => {
        if (valid) {
          this.$emit('submit', this.formData);
        }
      });
    }
  }
};
</script>
```

## 四、表单验证规则

### 4.1 动态生成验证规则

```javascript
// 动态生成验证规则
const rules = {};
schema.fields.forEach(field => {
  const fieldRules = [];
  
  // 必填验证
  if (field.required) {
    fieldRules.push({ required: true, message: `${field.label}不能为空` });
  }
  
  // 长度验证
  if (field.maxLength) {
    fieldRules.push({ 
      max: field.maxLength, 
      message: `${field.label}长度不能超过${field.maxLength}` 
    });
  }
  
  // 范围验证
  if (field.min && field.max) {
    fieldRules.push({ 
      type: 'number', 
      min: field.min, 
      max: field.max, 
      message: `${field.label}必须在${field.min}到${field.max}之间` 
    });
  }
  
  // 正则验证
  if (field.validate?.pattern) {
    fieldRules.push({ 
      pattern: new RegExp(field.validate.pattern), 
      message: `${field.label}格式不正确` 
    });
  }
  
  // 自定义验证
  if (field.validate?.validator) {
    fieldRules.push({ 
      validator: field.validate.validator, 
      trigger: 'blur' 
    });
  }
  
  if (fieldRules.length > 0) {
    rules[field.name] = fieldRules;
  }
});
```

### 4.2 自定义验证器

```javascript
// 自定义验证函数
function customValidator(rule, value, callback) {
  if (!value) {
    callback(new Error('字段不能为空'));
  } else if (value.length < 6) {
    callback(new Error('字段长度不能少于 6 个字符'));
  } else {
    callback();
  }
}

// 在 Schema 中使用
{
  "name": "password",
  "type": "input",
  "label": "密码",
  "validate": {
    "validator": customValidator
  }
}
```

## 五、表单联动逻辑

### 5.1 条件显示

```javascript
// 根据字段值控制显示隐藏
computed: {
  visibleFields() {
    return this.schema.fields.filter(field => {
      if (field.visibleWhen) {
        const { field: targetField, value } = field.visibleWhen;
        return this.formData[targetField] === value;
      }
      return true;
    });
  }
}
```


## 六、动态表单引擎面试题

### 1. 动态表单引擎的设计思路？

**参考答案：**

**核心思想：** 数据驱动 + 组件化

**设计要点：**
1. **JSON Schema 描述**：用 JSON 定义表单结构
2. **组件映射**：建立字段类型到 UI 组件的映射关系
3. **动态渲染**：根据 Schema 递归渲染组件树
4. **验证引擎**：动态生成验证规则

**优势：**
- 配置化，无需编码
- 灵活扩展
- 一次开发，多次复用

### 2. JSON Schema 如何设计？

**参考答案：**

**核心字段：**
- `name`: 字段名（唯一标识）
- `type`: 字段类型（input/select/radio 等）
- `label`: 字段标签
- `required`: 是否必填
- `options`: 选项数据（用于 select/radio/checkbox）
- `validate`: 验证规则

**扩展字段：**
- `visibleWhen`: 条件显示
- `defaultValue`: 默认值
- `placeholder`: 占位符
- `disabled`: 是否禁用

### 3. 如何实现表单的动态验证？

**参考答案：**

**实现方式：**
1. **规则配置**：在 JSON Schema 中定义验证规则
2. **动态生成**：根据配置生成验证规则对象
3. **框架集成**：利用 Element UI/Ant Design 的验证能力
4. **自定义验证**：支持传入验证函数

**验证时机：**
- 失焦时（blur）
- 输入时（change）
- 提交前（submit）

### 4. 前后端如何协同渲染表单？

**参考答案：**

**方案一：前端渲染**
- 后端返回 JSON Schema
- 前端解析并渲染
- 优点：交互体验好
- 缺点：前端复杂度高

**方案二：后端渲染**
- 后端生成完整 HTML
- 前端直接展示
- 优点：前端简单
- 缺点：灵活性差

**推荐方案：混合渲染**
- 结构由后端定义（Schema）
- 交互由前端处理
- 平衡灵活性和复杂度

### 5. 如何处理复杂的表单联动逻辑？

**参考答案：**

**联动类型：**
1. **显示联动**：根据 A 字段的值显示/隐藏 B 字段
2. **数据联动**：A 字段变化时，清空或更新 B 字段
3. **验证联动**：A 字段的值影响 B 字段的验证规则

**实现方式：**
- **配置式**：在 Schema 中定义 `visibleWhen`、`listenTo` 等属性
- **脚本式**：编写 JavaScript 函数处理联动逻辑
- **响应式**：利用 Vue/React 的响应式系统自动追踪依赖
